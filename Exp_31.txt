#include <stdio.h>

int main() {
    int pages[5] = {1,2,3,4,1};
    int frame[3] = {-1,-1,-1}, i, j, k = 0, faults = 0;

    for (i = 0; i < 5; i++) {
        int found = 0;
        for (j = 0; j < 3; j++)
            if (frame[j] == pages[i]) found = 1;

        if (!found) {
            frame[k] = pages[i];
            k = (k + 1) % 3;
            faults++;
        }
    }
    printf("Page Faults: %d\n", faults);
    return 0;
}
