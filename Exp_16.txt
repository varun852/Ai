#include <stdio.h>
#include <string.h>   // for strcpy()

struct emp {
    int id;
    char name[20];
};

int main() {
    struct emp e;
    FILE *fp;

    fp = fopen("emp.dat", "wb+");
    if (fp == NULL) {
        printf("File cannot be opened\n");
        return 1;
    }

    e.id = 1;
    strcpy(e.name, "Ravi");

    fwrite(&e, sizeof(e), 1, fp);

    rewind(fp);

    fread(&e, sizeof(e), 1, fp);
    printf("%d %s\n", e.id, e.name);

    fclose(fp);
    return 0;
}
