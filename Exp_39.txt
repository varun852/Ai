#include <stdio.h>
#include <stdlib.h>

int main() {
    int req[20], n, head, size, seek = 0;
    int i, j, temp;

    printf("Enter number of requests: ");
    scanf("%d", &n);

    printf("Enter disk size: ");
    scanf("%d", &size);

    printf("Enter request sequence: ");
    for (i = 0; i < n; i++)
        scanf("%d", &req[i]);

    printf("Enter head position: ");
    scanf("%d", &head);

    // Sort
    for (i = 0; i < n - 1; i++)
        for (j = i + 1; j < n; j++)
            if (req[i] > req[j]) {
                temp = req[i];
                req[i] = req[j];
                req[j] = temp;
            }

    seek += abs(head - size);
    seek += size;   // jump from end to start
    seek += abs(req[n-1] - 0);

    printf("Total Seek Time = %d\n", seek);
    return 0;
}
