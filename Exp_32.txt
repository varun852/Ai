#include <stdio.h>

int main() {
    int pages[5] = {1,2,3,1,4};
    int frame[3] = {-1,-1,-1}, used[3] = {0};
    int i, j, time = 0, faults = 0;

    for (i = 0; i < 5; i++) {
        int pos = -1;
        for (j = 0; j < 3; j++)
            if (frame[j] == pages[i]) pos = j;

        if (pos == -1) {
            int lru = 0;
            for (j = 1; j < 3; j++)
                if (used[j] < used[lru]) lru = j;
            frame[lru] = pages[i];
            used[lru] = ++time;
            faults++;
        } else used[pos] = ++time;
    }
    printf("Page Faults: %d\n", faults);
    return 0;
}
