#include <stdio.h>

int main() {
    int mem[5] = {100, 500, 200, 300, 600};
    int proc[4] = {212, 417, 112, 426};
    int i, j;

    for (i = 0; i < 4; i++) {
        for (j = 0; j < 5; j++) {
            if (mem[j] >= proc[i]) {
                printf("Process %d allocated to block %d\n", i+1, j+1);
                mem[j] -= proc[i];
                break;
            }
        }
    }
    return 0;
}
