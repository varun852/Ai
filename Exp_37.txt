#include <stdio.h>
#include <stdlib.h>

int main() {
    int n, i, head, seek = 0;
    int req[20];

    printf("Enter number of requests: ");
    scanf("%d", &n);

    printf("Enter request sequence: ");
    for (i = 0; i < n; i++)
        scanf("%d", &req[i]);

    printf("Enter initial head position: ");
    scanf("%d", &head);

    for (i = 0; i < n; i++) {
        seek += abs(head - req[i]);
        head = req[i];
    }

    printf("Total Seek Time = %d\n", seek);
    return 0;
}
